#include <stdio.h>
/*递归实现*/

void invert(int arr[], int k)
{
    if(k>1){
        invert(arr+1,k-2);
        int tmp = arr[0];
        arr[0] = arr[k-1];
        arr[k-1] = tmp;
    }
}
